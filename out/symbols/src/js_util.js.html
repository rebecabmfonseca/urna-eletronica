1 /**
  2  * Faz a chamada para obter ou mandar dados 
  3  * 
  4  * @param url
  5  * @param method
  6  * @param callback
  7  * 
  8  */
  9 function ajax(url, method, callback) {
 10   let request = new XMLHttpRequest();
 11   request.overrideMimeType("application/json");
 12   request.open(method, url, true);
 13   request.onreadystatechange = () => {
 14     if (request.readyState === 4 && request.status == "200") {
 15         callback(request.responseText);
 16     }
 17   };
 18   request.send(null);
 19 }
